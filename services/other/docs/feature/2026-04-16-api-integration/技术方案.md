# B组/C组接口联调 - 技术方案

## 背景与目标

**架构原则**：所有对 llm-py 的调用都必须经过 server-go（Go 层）代理，CLI 不能直接调用 llm-py。

```
CLI → server-go → llm-py
       ↑
   只调用这个
```

### 当前状态分析

| 层级 | 组件 | 状态 | 说明 |
|------|------|------|------|
| CLI | `nlp.ts` | ❌ 直接调用 llm-py | 违反架构原则 |
| CLI | `chat.ts` | ❌ 占位 | 未实现 |
| server-go | RAG 代理 | ✅ 已实现 | `rag_proxy.go`，代理 `/api/rag/*` 到 llm-py |
| server-go | Chat 代理 | ❌ 缺失 | 无 `/api/v1/chat` 路由 |
| server-go | NLP 代理 | ❌ 缺失 | 无 `/api/v1/nlp` 路由 |
| CLI | `status.ts` | ⚠️ 占位 | 显示 unknown，未调用 API |

### 问题诊断

1. **server-go 缺少 LLM 代理路由**：只有 RAG 代理，没有 Chat 和 NLP 代理
2. **CLI 违反架构原则**：`nlp.ts` 直接调用 `http://localhost:8000`，应改为调用 server-go
3. **CLI 命令未实现**：`/status`、`/chat`、KB/Timer/MCP 子命令都是占位代码

## 需求概述

### Phase A: server-go 新增代理路由

需要新增以下代理路由（server-go → llm-py）：

| 路由 | 方法 | 目标 | 说明 |
|------|------|------|------|
| `/api/v1/chat` | POST | `POST /api/chat` (llm-py) | 对话代理 |
| `/api/v1/chat/nlp` | POST | `POST /api/nlp` (llm-py) | NLP 意图识别代理 |

### Phase B: CLI API 调用重构

CLI 只调用 server-go，不直接调用 llm-py：

| CLI 调用 | 目标 | 说明 |
|---------|------|------|
| `GET /api/v1/health` | server-go | 健康检查 |
| `GET /api/v1/metrics` | server-go | 综合指标 |
| `POST /api/v1/chat` | server-go → llm-py | 对话（经 server-go 代理） |
| `POST /api/v1/chat/nlp` | server-go → llm-py | NLP 意图识别（经 server-go 代理） |
| `POST /api/v1/knowledge` | server-go | 知识库 CRUD |
| `POST /api/v1/knowledge/search` | server-go | 语义搜索（RAG） |
| `POST /api/v1/knowledge/ask` | server-go | RAG 问答 |
| `POST /api/v1/timers` | server-go | 定时任务 CRUD |
| `POST /api/v1/mcps` | server-go | MCP 配置 CRUD |

### Phase A: server-go 新增 LLM 代理路由

#### Step A1: 创建 LLM 代理服务
**目标文件**: `services/server-go/internal/application/service/llm_proxy.go`

```go
package service

import (
    "bytes"
    "context"
    "encoding/json"
    "net/http"
    "time"
)

const (
    defaultLLMTimeout = 60 * time.Second
    defaultLLMHost    = "http://localhost:8000"
)

// LLMProxy LLM 代理服务接口
type LLMProxy interface {
    Chat(ctx context.Context, req *ChatRequest) (*ChatResponse, error)
    ChatStream(ctx context.Context, req *ChatRequest) (<-chan string, error)
    NLP(ctx context.Context, req *NLPRequest) (*NLPResponse, error)
}

// llmProxy LLM 代理服务实现
type llmProxy struct {
    httpClient *http.Client
    baseURL    string
}

// ChatRequest 对话请求
type ChatRequest struct {
    Messages []Message `json:"messages"`
    Model    string    `json:"model,omitempty"`
    Stream   bool      `json:"stream,omitempty"`
}

// Message 消息
type Message struct {
    Role    string `json:"role"`
    Content string `json:"content"`
}

// ChatResponse 对话响应
type ChatResponse struct {
    Content string   `json:"content"`
    Usage   *Usage   `json:"usage,omitempty"`
}

// Usage token 使用量
type Usage struct {
    InputTokens  int `json:"input_tokens"`
    OutputTokens int `json:"output_tokens"`
}

// NLPRequest NLP 请求
type NLPRequest struct {
    Text    string                 `json:"text"`
    Context map[string]interface{} `json:"context,omitempty"`
}

// NLPResponse NLP 响应
type NLPResponse struct {
    Intent     string                 `json:"intent"`
    Confidence float64                `json:"confidence"`
    Entities   []NLPEntity            `json:"entities"`
    Parameters map[string]interface{} `json:"parameters"`
    Reply      string                 `json:"reply"`
}

// NLPEntity NLP 实体
type NLPEntity struct {
    Type  string `json:"type"`
    Value string `json:"value"`
}
```

#### Step A2: 创建 LLM 代理 Handler
**目标文件**: `services/server-go/internal/interfaces/http/handler/llm_handler.go`

```go
package handler

import (
    "net/http"

    "github.com/gin-gonic/gin"
    "github.com/ddo/server-go/internal/application/service"
)

// LLMHandler LLM 请求处理
type LLMHandler struct {
    llmProxy service.LLMProxy
}

// NewLLMHandler 创建 LLM 处理器
func NewLLMHandler(llmProxy service.LLMProxy) *LLMHandler {
    return &LLMHandler{llmProxy: llmProxy}
}

// Chat Chat 对话
func (h *LLMHandler) Chat(c *gin.Context) {
    var req service.ChatRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    resp, err := h.llmProxy.Chat(c.Request.Context(), &req)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }

    c.JSON(http.StatusOK, resp)
}

// ChatStream 流式对话
func (h *LLMHandler) ChatStream(c *gin.Context) {
    var req service.ChatRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    c.Writer.Header().Set("Content-Type", "text/event-stream")
    c.Writer.Header().Set("Cache-Control", "no-cache")
    c.Writer.Header().Set("Connection", "keep-alive")

    stream, err := h.llmProxy.ChatStream(c.Request.Context(), &req)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }

    for chunk := range stream {
        c.Writer.Write([]byte("data: " + chunk + "\n\n"))
        c.Writer.Flush()
    }
    c.Writer.Write([]byte("data: [DONE]\n\n"))
}

// NLP NLP 意图识别
func (h *LLMHandler) NLP(c *gin.Context) {
    var req service.NLPRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    resp, err := h.llmProxy.NLP(c.Request.Context(), &req)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
        return
    }

    c.JSON(http.StatusOK, resp)
}
```

#### Step B1: 创建统一 API 客户端
**目标文件**: `services/cli/src/services/api-client.ts`

```typescript
/**
 * API Client
 * 统一调用 server-go API（server-go 再代理到 llm-py）
 */

interface ApiClientConfig {
  serverGoUrl: string;  // http://localhost:8080
  timeout: number;
}

export function createApiClient(config: ApiClientConfig) {
  const { serverGoUrl, timeout } = config;
  const base = serverGoUrl;

  // 通用请求方法
  async function request<T>(path: string, options?: RequestInit): Promise<T> {
    const url = `${base}${path}`;
    const response = await fetch(url, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        ...options?.headers,
      },
    });
    if (!response.ok) {
      throw new Error(`API ${path} failed: ${response.status}`);
    }
    return response.json() as T;
  }

  // === 健康检查 ===
  async function getHealth(): Promise<ServerGoHealth>
  async function getMetrics(): Promise<Metrics>

  // === LLM（经 server-go 代理到 llm-py）===
  async function chat(messages: Message[], stream?: boolean): Promise<ChatResponse>
  async function analyzeText(text: string, context?: Record<string, unknown>): Promise<NLPResponse>

  // === 知识库 ===
  async function getKnowledgeList(params): Promise<KnowledgeListResponse>
  async function createKnowledge(data): Promise<KnowledgeResponse>
  async function deleteKnowledge(uuid: string): Promise<void>
  async function searchKnowledge(query, top_k?): Promise<SearchResponse>
  async function askKnowledge(question): Promise<AskResponse>

  // === 定时任务 ===
  async function getTimers(): Promise<TimerListResponse>
  async function createTimer(data): Promise<TimerResponse>
  async function pauseTimer(uuid): Promise<void>
  async function resumeTimer(uuid): Promise<void>
  async function deleteTimer(uuid): Promise<void>

  // === MCP ===
  async function getMcpList(): Promise<McpListResponse>
  async function createMcp(data): Promise<McpResponse>
  async function testMcp(uuid): Promise<McpTestResponse>
  async function deleteMcp(uuid): Promise<void>

  return {
    getHealth, getMetrics,
    chat, analyzeText,
    getKnowledgeList, createKnowledge, deleteKnowledge,
    searchKnowledge, askKnowledge,
    getTimers, createTimer, pauseTimer, resumeTimer, deleteTimer,
    getMcpList, createMcp, testMcp, deleteMcp,
  };
}

/** 全局 API Client 实例 */
let globalClient: ReturnType<typeof createApiClient> | null = null;

export function getApiClient(config?: { dataDir?: string }): ReturnType<typeof createApiClient> {
  if (!globalClient) {
    // 从 config.yaml 读取 endpoints.serverGo
    const serverGoUrl = process.env.DDO_SERVER_GO_URL || 'http://localhost:8080';
    globalClient = createApiClient({
      serverGoUrl,
      timeout: 30000,
    });
  }
  return globalClient;
}

export function resetApiClient(): void {
  globalClient = null;
}
```
